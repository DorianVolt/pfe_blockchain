\begin{frame}{Spécificités d'une blockchain}
    \begin{itemize}
        \item Aucune autorité centrale.
        \item Abscence de hiérarchie.
        \item Immuable.
        \item Aucune présence d'informations contradictoires.
        \item Notion de partage et consensus. 
    \end{itemize}
\end{frame}
